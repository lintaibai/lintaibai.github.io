вқӨReact-ReactRouterи·Ҝз”ұзҡ„дҪҝз”Ё вҖӢ

React Router еңЁ 2021 е№ҙ 11 жңҲд»Ҫзҡ„ж—¶еҖҷжӣҙж–° v6 зҡ„зүҲжң¬пјҢжң¬ж¬Ўдё»иҰҒеӯҰд№ зҡ„д№ҹжҳҜv6пјҢйҮҢйқўжӣҙж”№дәҶеҫҲеӨҡж–°еҶҷжі•зү№жҖ§пјҒ

1гҖҒи·Ҝз”ұзҡ„и®ӨиҜҶ вҖӢ

е®ҳзҪ‘з»ҷд»–е®ҡд№үжҳҜдёҖдёӘеә“пјҢдёҚжҳҜдёҖдёӘжЎҶжһ¶гҖӮ

Reactе®ҳж–№е№¶жІЎжңүжҸҗдҫӣи·Ҝз”ұеә“пјҢи·Ҝз”ұжҸ’д»¶,иҖҢжҳҜжҺЁиҚҗжҲ‘д»¬дҪҝз”ЁReactз»“еҗҲReactRouterжқҘиҝӣиЎҢи·Ҝз”ұзҡ„жҳ е°„гҖӮ

жӯЈеёёзҡ„и·Ҝз”ұејҖеҸ‘жЁЎејҸеҲҶдёәд»ҘдёӢдёүз§Қпјҡ

  • зј–зЁӢејҸи·Ҝз”ұпјҡиҮӘе·ұеңЁйЎөйқўдёӯеҶҷд»Јз ҒиҝӣиЎҢи·Ҝз”ұзј–зЁӢпјҢеҢ…жӢ¬жҳ е°„гҖҒи·Ҝз”ұеј•е…ҘгҖӮ
  • й…ҚзҪ®ејҸи·Ҝз”ұпјҡе°Ҷи·Ҝз”ұжҳ е°„еҶҷеҲ°й…ҚзҪ®ж–Үд»¶дёӯпјҢеү©дёӢзҡ„е°ұжҳҜиҮӘеҠЁжёІжҹ“гҖӮ
  • зәҰе®ҡејҸи·Ҝз”ұпјҡж— йңҖй…ҚзҪ®и·Ҝз”ұпјҢдҪ зҡ„йЎ№зӣ®ж–Үд»¶зі»з»ҹе°ұжҳҜи·Ҝз”ұгҖӮпјҲNextе°ұжҳҜиҝҷз§Қж–№ејҸпјү

е…ҲжқҘзңӢзңӢVuejsй…ҚзҪ®ејҸи·Ҝз”ұпјҡ

js

import VueRouter from "vue-router"

  {
       path:"/login",
       name:"Login"
       component:Login
  }

   routes,
   mode:"hash"
})

export default router

4гҖҒи·Ҝз”ұиҜҰз»ҶиҜҙжҳҺпјҡ вҖӢ

и·Ҝз”ұеӨ§иҮҙи®ӨиҜҶ вҖӢ

и·Ҝз”ұеҷЁзӣ®еүҚжңүдёӨз§ҚпјҲпјҡBrowserRouterгҖҒHashRouterпјүпјҢеҶіе®ҡдәҶеҪ“еүҚи·Ҝз”ұзҡ„жЁЎејҸгҖӮ

BrowserRouterпјҡй»ҳи®ӨйҮҮз”ЁhistoryжЁЎејҸ

HashRouterпјҡйҮҮз”ЁhashжЁЎејҸгҖӮи·Ҝеҫ„и®ҝй—®зҡ„ж—¶еҖҷйңҖиҰҒ/#/

и·Ҝз”ұжҳ е°„пјҡRoutesгҖҒRouteдё»иҰҒиҙҹиҙЈиҝӣиЎҢи·Ҝз”ұи·Ҝеҫ„еҢ№й…ҚпјҢжҸҗдҫӣжёІжҹ“зҡ„з»„д»¶

RoutesпјҡиЎЁзӨәеҸҜд»ҘеҢ…еҗ«еӨҡдёӘжҳ е°„и§„еҲҷгҖӮд»ҺдёҠеҲ°е°Ҹзҡ„иҝӣиЎҢеҢ№й…ҚгҖӮеҪ“еҢ№й…ҚжҲҗеҠҹз»“жқҹеҢ№й…Қ

RouteпјҡиҝӣиЎҢи·Ҝз”ұжҳ е°„пјҢpathи·Ҝз”ұи·Ҝеҫ„пјҢelementжҸҗдҫӣжҳ е°„з»„д»¶гҖӮжҢҮе®ҡзҡ„иҝҷдёӘең°ж–№жёІжҹ“з»„д»¶

и·Ҝз”ұеҜјиҲӘпјҡLinkгҖҒNavLink

йҖҡиҝҮLinkе’ҢNavLinkз»„д»¶еҸҜд»Ҙе®һзҺ°и·Ҝз”ұзҡ„еҲҮжҚўпјҢзұ»дјјдәҺVueд№Ӣдёӯзҡ„router-linkиҝӣиЎҢеҜ№еҸ·е…Ҙеә§пјҒ

и·Ҝз”ұжҳ е°„и§„еҲҷ вҖӢ

й»ҳи®Өзҙўеј•е’ҢйҮҚе®ҡеҗ‘ йңҖжұӮпјҡеҪ“жҲ‘д»¬зӮ№еҮ»иҝӣжқҘд»ҘеҗҺйҰ–е…Ҳиҝӣе…Ҙзҡ„жҳҜжҲ‘д»¬зҡ„homeдё»йЎөйқўпјҡindex зҡ„дҪңз”Ё

6гҖҒйҮҚе®ҡеҗ‘и·Ҝз”ұredirect(Navigate) вҖӢ

жҺҘдёӢжқҘжҲ‘д»¬иҫ“е…Ҙhttp://localhost:3000/a еҚҙеҸ‘зҺ°дёҖзүҮз©әзҷҪпјҢйӮЈд№ҲжҲ‘д»¬жғіиҰҒиҫ“е…ҘдёҚеҢ№й…Қзҡ„зӣҙжҺҘи·іиҪ¬еҲ°жҲ‘д»¬зҡ„йҰ–йЎөеә”иҜҘеҰӮдҪ•еҒҡе‘ў

е…ҲзңӢзңӢд№ӢеүҚv5зүҲжң¬зҡ„йҮҚе®ҡеҗ‘еҰӮдҪ•еҒҡзҡ„

еңЁжҲ‘д»¬и®ҝй—®'/' зҡ„ж—¶еҖҷзӣҙжҺҘе°ұйҮҚе®ҡеҗ‘еҲ°дәҶвҖҳ/filemsвҖҷ иҝҷдёӘи·Ҝз”ұ

еңЁиҝҷйҮҢзЎ®е®һйҒҮеҲ°дәҶеқ‘пјҢе°ұжҳҜеҶҷзҡ„е®Ңе…ЁдёҖж ·пјҢдҪҶжҳҜе°ұжҳҜдёҚз”ҹж•ҲпјҢеүҚз«Ҝзһ¬жҒҜдёҮеҸҳпјҢејәзғҲе»әи®®жңүдәӣдёңиҘҝзңӢзңӢдјҡдәҶе°ұиЎҢпјҢи®°дҪҸдәҶд№ҹжІЎе•Ҙз”ЁпјҢжңҖйҮҚиҰҒзҡ„е°ұжҳҜзҗҶи§Је№¶дё”дјҡжҹҘйҳ…зӣёе…ізҡ„ж–ҮжЎЈпјҢиҝҷйҮҢж–°зүҲжң¬зҡ„v6еҶҷжі•е·Із»ҸеҸҳжҲҗдәҶдёӢйқўиҝҷж ·еӯҗпјҡпјҲеҗ«д№үе°ұжҳҜй«ҳж”№еҸҳдәҶд»ҘеҗҺ v6 дёӯзҡ„йҮҚе®ҡеҗ‘жӣҙеҠ з®ҖжҙҒе’Ңзӣҙи§Ӯпјү

8гҖҒReact.lazy дјҳеҢ–и·Ҝз”ұ вҖӢ

и®ӨиҜҶ вҖӢ

жҲ‘д»¬еҸ‘зҺ°еҠ дәҶи·Ҝз”ұд»ҘеҗҺжҲ‘д»¬зҡ„еҠ иҪҪзј“ж…ўдәҶеҫҲеӨҡпјҢReact.lazyеҸҜд»ҘеҮҸе°‘жҲ‘д»¬еә”з”Ёзҡ„еҲқе§ӢеҠ иҪҪж—¶й—ҙпјҢжҸҗеҚҮз”ЁжҲ·дҪ“йӘҢпјҢеҜ№дәҺеӨ§еһӢеҚ•йЎөеә”з”ЁпјҲSPAпјүжҲ–йңҖиҰҒдјҳеҢ–еҠ иҪҪж—¶й—ҙзҡ„еә”з”Ёзү№еҲ«жңүз”Ё

React.lazy жҳҜ React жҸҗдҫӣзҡ„дёҖдёӘеҠЁжҖҒеҜје…Ҙз»„д»¶зҡ„ж–№жі•пјҢдҪңз”ЁжҳҜйңҖиҰҒж—¶жүҚеҠ иҪҪз»„д»¶пјҢиҖҢдёҚжҳҜеңЁеә”з”ЁеҲқе§ӢеҢ–ж—¶е°ұеҠ иҪҪжүҖжңүз»„д»¶пјҢд»ҺиҖҢжҸҗй«ҳеә”з”Ёзҡ„жҖ§иғҪе’ҢеҠ иҪҪйҖҹеәҰгҖӮ

React.lazy еҮҪж•°жҺҘеҸ—дёҖдёӘеҮҪж•°дҪңдёәеҸӮж•°пјҢеҠЁжҖҒеҜје…Ҙз»„д»¶гҖӮеҪ“з»„д»¶йңҖиҰҒиў«жёІжҹ“ж—¶пјҢReact е°Ҷи°ғз”ЁиҝҷдёӘеҮҪж•°жқҘеҠ иҪҪз»„д»¶гҖӮеҠ иҪҪе®ҢжҲҗеҗҺпјҢз»„д»¶е°Ҷиў«зј“еӯҳпјҢд»ҘдҫҝеңЁе°ҶжқҘзҡ„жёІжҹ“дёӯйҮҚз”ЁгҖӮ

React.lazy з»“еҗҲ Suspense з»„д»¶дҪҝз”Ёж—¶пјҢеҸҜд»Ҙдјҳйӣ…ең°еӨ„зҗҶеҠЁжҖҒеҠ иҪҪз»„д»¶ж—¶зҡ„ loading зҠ¶жҖҒпјҢдҪҝд»Јз ҒжӣҙеҠ жё…жҷ°е’Ңжҳ“дәҺз»ҙжҠӨгҖӮ

import { BrowserRouter as Router, Route, Switch } from 'react-router-dom';

const Admin = lazy(() => import('./Admin'));

  return (
    <Router>
      <Suspense fallback={<div>Loading...</div>}>
        <Switch>
          <Route exact path="/" component={Login} />
          <Route path="/admin" component={Admin} />
        </Switch>
      </Suspense>
    </Router>
  );
};

export default App;
